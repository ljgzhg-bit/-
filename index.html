        // --- JavaScript 逻辑区域 ---

 

        function checkDate() {

            const dateInput = document.getElementById('dateInput').value;

            

            // 获取输入框的值，格式通常是 YYYY-MM-DD

            // 我们检查字符串是否以 "-02-02" 结尾，这样就可以忽略年份

            if (dateInput && dateInput.endsWith('-02-02')) {

                

                // 1. 先触发爱心特效

                triggerHeartRain();

                }, 800);

 

            } else {

                // 如果日期不对

                alert('��� 日期好像不太对哦，再想想？提示：是二月里的某一天。');

            }

        }

 

        // 生成爱心雨的函数

        function triggerHeartRain() {

            const container = document.querySelector('.hearts-container');

            const heartCount = 40; // 生成爱心的数量

 

            for (let i = 0; i < heartCount; i++) {

                setTimeout(() => {

                    const heart = document.createElement('div');

                    heart.classList.add('heart');

                    // 可以随机选择不同的爱心表情

                    heart.innerHTML = ['���', '���', '���', '���'][Math.floor(Math.random() * 4)];

                    

                    // 随机位置、大小和飘动速度

                    heart.style.left = Math.random() * 100 + 'vw';

                    heart.style.fontSize = (Math.random() * 1.5 + 1) + 'rem';

                    heart.style.animationDuration = (Math.random() * 3 + 3) + 's'; // 3到6秒之间

                    

                    container.appendChild(heart);

 

                    // 动画结束后移除元素，清理内存

                    setTimeout(() => { heart.remove(); }, 6000);

                }, i * 100); // 每隔100ms生成一个，错落有致

            }

        }
